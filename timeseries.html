First, fit an ARIMA model to the July daily data seperately for each analyte and station. Note that turbidity and fluorescence are log-transformed for the model:

ts.data = action.daily %>%
  mutate(
    Station = factor(Station, c("(C-2B)  Collinsville B", "(S-64)  National Steel", "(S-54)  Hunter Cut")),
    ModelValue = if_else(str_detect(Analyte, "Turb|Flu"),
      log(Mean), Mean)
  ) %>%
  group_by(Station, Analyte) %>%
  mutate(
    group.mean = mean(ModelValue),
    modelval = ModelValue - group.mean
  ) %>%
  ungroup()

gm = ts.data %>%
  select(Station, Analyte, group.mean) %>%
  distinct()

ts.jul = ts.data %>%
  as_tsibble(index = "Datetime", key = c("Station", "Analyte")) %>%
  filter(near(month(Datetime), 7)) %>%
  model(julmodel = ARIMA(modelval))

We then apply these models to the full July-September dataset using the refit function. We set the parameter reestimate = FALSE to use the same coefficients as determined using the July data:

ts.all = ts.data %>%
  group_nest(Station, Analyte) %>%
  left_join(ts.jul, by = c("Station", "Analyte")) %>%
  mutate(all.model = map2(data, julmodel, ~ refit(.y,
    new_data = as_tsibble(.x, index = "Datetime"),
    reestimate = FALSE))) %>%
    transmute(Station, Analyte, tidy = map(all.model, augment)) %>%
  unnest() %>%
  left_join(gm) %>%
  mutate(newval = .resid + group.mean)
## Joining, by = c("Station", "Analyte")

We then fit linear models of Month and Station for each analyte:

newmodels = ts.all %>%
  mutate(
    Month = factor(month(Datetime), 7:9, month.abb[7:9]),
    Station = factor(Station)
  ) %>%
  group_nest(Analyte) %>%
  mutate(model = map(data, ~ lm(newval~ Month * Station, data = .x)))
